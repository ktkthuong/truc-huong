import React, { Component } from 'react'

export default class UpComponent extends Component {
    constructor(props){
        super(props);
       
        this.state={
            count:1
        }
    }
    
  render() {
    return (
    //   <div>UpComponent</div>
    <div>
    <p>Count: {this.state.count}</p>
    <button type='button' onClick={() => this.setState({count: this.state.count+1})}>Add</button>
    
    </div>
    )
  }
}
